94 static tuple<Polynomial<T>, Polynomial<T>, Polynomial<T>> extended_gcd(

97 if(b.degree() < 0) {

101 auto [g, x, y] = extended_gcd(b, r);

176template<typename T>
177bool is_irreducible(const Polynomial<T>& poly) {
178 int deg = poly.degree();
179 if(deg <= 0)
180 return false;
181 if(deg == 1)
182 return true;
183 int field_mod = T::MOD_VALUE;
184 for (int d = 1; d <= deg / 2; d++) {
185 vector<T> coeffs(d, T(0));
186 function<bool(int)> gen = [&](int pos) -> bool {
187 if (pos == d) {
188 vector<T> candidate_coeffs = coeffs;
189 candidate_coeffs.push_back(T(1)); // делаем ведущий коэффициент равным 1
190 Polynomial<T> candidate(candidate_coeffs);
191 auto rem = poly.divmod(candidate).second;
192 if(rem.degree() < 0)
193 return true; // найден делитель -> многочлен приводим.
194 return false;
195 }
196 for (int i = 0; i < field_mod; i++) {
197 coeffs[pos] = T(i);
198 if (gen(pos + 1))
199 return true;
200 }
201 return false;
202 };
203 if(gen(0))
204 return false;
205 }
206 return true;
207}
