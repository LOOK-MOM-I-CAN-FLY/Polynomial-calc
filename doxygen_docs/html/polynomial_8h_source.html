156 pair<Polynomial, Polynomial> divmod(const Polynomial& divisor) const {
157 if(divisor.coeffs.empty())
158 throw runtime_error("Division by zero polynomial");
159 Polynomial dividend = *this;
160 Polynomial quotient;
161 quotient.coeffs.resize(max(0, dividend.degree() - divisor.degree() + 1), T(0));
162 while(dividend.degree() >= divisor.degree() && dividend.degree() >= 0) {
163 int diff = dividend.degree() - divisor.degree();
164 T factor = dividend.coeffs.back() / divisor.coeffs.back();
165 vector<T> term(diff+1, T(0));
166 term[diff] = factor;
167 Polynomial<T> monomial(term);
168 quotient.coeffs[diff] = factor;
169 dividend = dividend - (divisor * monomial);
170 dividend.normalize();
171 }
172 quotient.normalize();
173 dividend.normalize();
174 return {quotient, dividend};
175 }
